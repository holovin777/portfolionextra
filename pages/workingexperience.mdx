import {useSSG} from 'nextra/ssg';
import {Tab, Tabs} from 'nextra-theme-docs';

export const getStaticProps = ({params}) => {
    return fetch(`http://localhost:8080/api/v1/customer/1adf2007-ba51-42ae-b911-4c51e05e1b18`)
        .then(res => res.json())
        .then(customer => ({
            props: {
                ssg: {
                    customer: customer
                }
            },
            revalidate: 60
        }))
}

export const WorkingExperience = () => {
    const {customer} = useSSG();
    const tabItems = [];
    customer.workingExperiences.map(workingExperience =>
    {tabItems.push(workingExperience.startedWork)});
    return (
        <div>
            <Tabs items={tabItems}>
                {customer.workingExperiences.map(workingExperience =>
                    <Tab key={workingExperience.id}>
                        <div><h2>🏭 <i>{workingExperience.company.name}</i></h2></div>
                        <div><h3>👷🏻 {workingExperience.positionAtWork.name}</h3></div>
                        <hr></hr>
                        <div>🗺 {workingExperience.company.location}</div>
                        <hr></hr>
                        {workingExperience.finishedWork !== null &&
                            <div>Finished worked in {workingExperience.finishedWork}</div>
                        }
                    </Tab>
                )
                }
            </Tabs>
        </div>
    )
}
