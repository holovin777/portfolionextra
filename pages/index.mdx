import {useSSG} from 'nextra/ssg';

export const getStaticProps = ({params}) => {
    return fetch(`http://localhost:8080/api/v1/customer/d2cf7fc0-fc7e-4c59-a111-77ccd4f008fa`)
        .then(res => res.json())
        .then(customer => ({
            props: {
                ssg: {
                    customer: customer
                }
            },
            revalidate: 60
        }))
}

export const ContactIformation = () => {
    const {customer} = useSSG()
    return (
        <div>
            <div>
                <h3>{customer.firstName} {customer.lastName}</h3>
            </div>
            <div>
                {customer.phoneNumber}

            </div>
            <div>
                {customer.email}
            </div>
            {customer.website !== null &&
                <div>
                    <a href={customer.website}>{customer.website}</a>
                </div>
            }
            <div>
                {customer.residence}
            </div>
        </div>
    )
}
