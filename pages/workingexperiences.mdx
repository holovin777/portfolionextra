import {useSSG} from 'nextra/ssg';
import {Tab, Tabs} from 'nextra-theme-docs';

export const getStaticProps = ({params}) => {
    return fetch(`http://localhost:8080/api/v1/customer/d2cf7fc0-fc7e-4c59-a111-77ccd4f008fa/working-experience/all`)
        .then(res => res.json())
        .then(workingExperiences => ({
            props: {
                ssg: {
                    workingExperiences: workingExperiences
                }
            },
            revalidate: 60
        }))
}

export const WorkingExperiences = () => {
    const {workingExperiences} = useSSG();
    const tabItems = [];
    workingExperiences.map(workingExperience =>
    {tabItems.push(workingExperience.startedWork)});
    return (
        <div>
            <hr></hr>
            Started work:
            <Tabs items={tabItems}>
                {workingExperiences.map(workingExperience =>
                    <Tab key={workingExperience.id}>
                        <div><h2>{workingExperience.company.name}</h2></div>
                        <div><h3>{workingExperience.positionAtWork.name}</h3></div>
                        <hr></hr>
                        <div>ðŸ—º {workingExperience.company.location}</div>
                        <hr></hr>
                        {workingExperience.finishedWork !== null &&
                            <div>Finished worked in {workingExperience.finishedWork}</div>
                        }
                    </Tab>
                )
                }
            </Tabs>
        </div>
    )
}
