import {useSSG} from 'nextra/ssg';
import {Tab, Tabs} from 'nextra-theme-docs';

export const getStaticProps = ({params}) => {
    return fetch(`http://localhost:8080/api/v1/customer/e28e100f-b2dd-49c7-b44a-81fb39cc249e/certificate/all`)
        .then(res => res.json())
        .then(certificates => ({
            props: {
                ssg: {
                    certificates: certificates
                }
            },
            revalidate: 60
        }))
}

export const Certificate = () => {
    const {certificates} = useSSG();
    const tabItems = [];
    certificates.map(certificate =>
    {tabItems.push(certificate.completedDate)});
    return (
        <div>
            <hr></hr>
            Completed date:
            <Tabs items={tabItems}>
                {certificates.map(certificate =>
                    <Tab key={certificate.id}>
                        <div><h2>{certificate.name}</h2></div>
                        <div><h3>{certificate.brandName}</h3></div>
                    </Tab>
                )
                }
            </Tabs>
        </div>
    )
}
